    public < N extends Integer> Integer addOperations(N val1, N val2)
    {
        //N result = val1 + val2;
        return val1.intValue() + val2.intValue();
    }

    public < N extends Integer> Integer multiplyOperations(N val1, N val2)
    {
        return val1.intValue() * val2.intValue();
    }
